36// Butterworth Low Pass Filter (LPF) class template
37// Order: 1 to 4
38template <order Order, typename T = float>
39class butterworth_lpf
40{
41 static constexpr order ORDER = Order;
42
43 static constexpr float PI = math_constants<T>::pi;
44
45 T _wc = T{0}; // [rad/s] Cut-off frequency
46 T _t = T{1}; // [s] Sample time, default to 1 second

63 constexpr void pre_warp()
64 {
65 // Check that fc <= fs / 2 (Nyquist)
66 const T fc = clamp<T>(this->_fc, T{0}, this->_fs / T{2});
67
68 this->_t = T{1} / this->_fs;
69
70 // Computes the pre-warped cut-off frequency (`_wc`) in rad/s.
71 this->_wc = (T{2} / this->_t) * tanf(PI * fc / this->_fs);
72 }
73
74 public:
76 constexpr auto value() const -> T { return this->_y[0]; }

79 constexpr void reset()

81 this->_u.fill(0.f);
82 this->_y.fill(0.f);
83 }

87 constexpr auto update(const T val) -> T
88 {
89 // Shift samples
90 for (size_t i = static_cast<size_t>(ORDER); i > 0; --i)
91 {
92 this->_u[i] = this->_u[i - 1];
93 this->_y[i] = this->_y[i - 1];
94 }
95
96 // Set new input
97 this->_u[0] = val;
98 this->_y[0] = T{0};
99
100 // This loop calculates the output value (`_y[0]`) of the filter by
101 // applying the filter coefficients to the input (`_u`) and output
102 // (`_y`) buffers.
103 //
104 // - The numerator coefficients (`_num`) are multiplied with the input
105 // values to contribute to the output.
106 // - The denominator coefficients (`_den`) are multiplied with the past
107 // output values (for `i > 0`) and subtracted to account for feedback.
108 //
109 // The result is stored in `_y[0]`, representing the current filtered
110 // output.
111 for (size_t i = 0; i <= static_cast<size_t>(ORDER); ++i)
112 {
113 this->_y[0] += this->_num[i] * this->_u[i];
114
115 if (i > 0)
116 {
117 this->_y[0] -= this->_den[i] * this->_y[i];
118 }
119 }
120
121 this->_y[0] /= this->_den[0];
122 return this->_y[0];
123 }

129 constexpr butterworth_lpf(T fc, T fs) : _fc(fc), _fs(fs)
130 {
131 this->pre_warp();
132 this->reset();
133
134 // Set filter numerator and denominator coefficients
135
136 // Pre-warped cut-off frequency in seconds
137 const float wc_t = this->_wc * this->_t;
138 // Pre-warped cut-off frequency squared
139 const float wc_tsq = wc_t * wc_t;
140 // Pre-warped cut-off frequency cubed
141 const float wc_tcu = wc_t * wc_tsq;
142 // Pre-warped cut-off frequency to the fourth power
143 const float wc_tfo = wc_tsq * wc_tsq;
144
145 if constexpr (ORDER == order::first)
146 {
147 this->_num[0] = wc_t;
148 this->_num[1] = wc_t;
149
150 this->_den[0] = 2 + wc_t;
151 this->_den[1] = -2 + wc_t;
152 }
153 else if constexpr (ORDER == order::second)
154 {
155 this->_num[0] = wc_tsq;
156 this->_num[1] = 2 * wc_tsq;
157 this->_num[2] = wc_tsq;
158
159 this->_den[0] = 4 + 2.8284 * wc_t + wc_tsq;
160 this->_den[1] = -8 + 2 * wc_tsq;
161 this->_den[2] = 4 - 2.8284 * wc_t + wc_tsq;
162 }
163 else if constexpr (ORDER == order::third)
164 {
165 this->_num[0] = wc_tcu;
166 this->_num[1] = 3 * wc_tcu;
167 this->_num[2] = 3 * wc_tcu;
168 this->_num[3] = wc_tcu;
169
170 this->_den[0] = 8 + 8 * wc_t + 4 * wc_tsq + wc_tcu;
171 this->_den[1] = -24 - 8 * wc_t + 4 * wc_tsq + 3 * wc_tcu;
172 this->_den[2] = 24 - 8 * wc_t - 4 * wc_tsq + 3 * wc_tcu;
173 this->_den[3] = -8 + 8 * wc_t - 4 * wc_tsq + wc_tcu;
174 }
175 else if constexpr (ORDER == order::fourth)
176 {
177 this->_num[0] = wc_tfo;
178 this->_num[1] = 4 * wc_tfo;
179 this->_num[2] = 6 * wc_tfo;
180 this->_num[3] = 4 * wc_tfo;
181 this->_num[4] = wc_tfo;
182
183 constexpr float ALPHA = 2.6132;
184 constexpr float BETA = 3.41430612;
185
186 this->_den[0] = 16 + 8 * ALPHA * wc_t + 4 * BETA * wc_tsq + 2 * ALPHA * wc_tcu + wc_tfo;
